index.html:

<html>
<head>
    <title><%= title %></title>

app.js (node.js):


const express = require('express');
const app = express();
const ejs = require('ejs');
const sqlite3 = require('sqlite3').verbose();

// SQLite DB 연결
const db = new sqlite3.Database('mydb.db', (err) => {
  if (err) {
    console.error(err.message);
  }
  console.log('Connected to the database.');
});

// 루트 경로에서 index.ejs 렌더링
app.get('/', (req, res) => {
  // SQLite DB에서 데이터 조회
  db.get('SELECT * FROM mytable', (err, row) => {
    if (err) {
      console.error(err.message);
      return res.status(500).send('Internal Server Error');
    }

    // index.ejs 렌더링 및 데이터 전달
    ejs.renderFile('index.ejs', {
      title: row.title,
      description: row.description,
      author: row.author
    }, (err, renderedHtml) => {
      if (err) {
        console.error(err.message);
        return res.status(500).send('Internal Server Error');
      }
      res.send(renderedHtml);
    });
  });
});

// 서버 시작
const port = 3000;
app.listen(port, () => {
  console.log(`Server is running on http://localhost:${port}`);
});

위의 코드 예시는 다음과 같은 과정을 수행합니다:

  1. index.html은 EJS 문법(<%= %>)을 사용하여 동적으로 제목, 설명, 제작자 정보를 표시하는 간단한 HTML 템플릿입니다.
  2. app.js는 Express와 EJS를 사용하여 웹 서버를 구성합니다.
  3. SQLite 데이터베이스를 연결하고, mytable에서 제목, 설명, 제작자 정보를 가져옵니다.
  4. 웹 서버에 들어오는 루트 경로 요청(http://localhost:3000/)에 대해 index.ejs 파일을 렌더링하여 클라이언트에 전달합니다.
  5. index.ejs는 SQLite DB에서 조회한 값으로 제목, 설명, 제작자 정보를 출력하고, 브라우저로 전달합니다.

이 예시 코드를 실행하려면 먼저 express, ejs, sqlite3 라이브러리를 설치해야 합니다. 프로젝트 폴더에서 아래 명령을 실행하세요:


npm install express ejs sqlite3

그리고 index.html과 app.js 파일을 프로젝트 폴더에 생성한 후, SQLite DB 파일(mydb.db)을 생성하여 데이터를 채워넣어야 합니다. 데이터베이스 생성 및 테이블 작성은 이 예시의 범위를 벗어나므로 자세한 내용은 생략하겠습니다.

이후 아래 명령을 실행하여 서버를 실행합니다:


node app.js

웹 브라우저에서 http://localhost:3000/로 접속하면 index.html에 작성된 내용과 SQLite DB에서 조회한 값이 출력되는 페이지를 확인할 수 있습니다.

아래는 Node.js에서 SQLite를 사용하여 데이터베이스를 생성하고 테이블을 만들고 데이터를 추가하는 예시 코드입니다. Node.js에서 SQLite를 사용하려면 먼저 'sqlite3' 모듈을 설치해야 합니다.


npm install sqlite3


const sqlite3 = require('sqlite3').verbose();

// 데이터베이스 파일 생성 및 연결
const db = new sqlite3.Database('mydb.db', (err) => {
  if (err) {
    console.error(err.message);
    return;
  }
  console.log('Connected to the database.');

  // 테이블 생성
  db.run(`
    CREATE TABLE IF NOT EXISTS mytable (
      id INTEGER PRIMARY KEY,
      title TEXT,
      description TEXT,
      author TEXT
    )
  `, (err) => {
    if (err) {
      console.error(err.message);
      return;
    }
    console.log('Table "mytable" created.');

    // 데이터 추가
    const title = 'Sample Title';
    const description = 'This is a sample description.';
    const author = 'John Doe';

    db.run('INSERT INTO mytable (title, description, author) VALUES (?, ?, ?)',
      [title, description, author], (err) => {
        if (err) {
          console.error(err.message);
          return;
        }
        console.log('Data inserted successfully.');

        // 데이터베이스 연결 종료
        db.close((err) => {
          if (err) {
            console.error(err.message);
            return;
          }
          console.log('Database connection closed.');
        });
      });
  });
});

위의 코드는 다음과 같은 작업을 수행합니다:

  1. 'mydb.db'라는 SQLite 데이터베이스 파일을 생성하고 연결합니다.
  2. 'mytable'이라는 테이블을 생성합니다. 해당 테이블에는 id(자동 생성된 고유 번호), title, description, author라는 컬럼이 있습니다.
  3. 'mytable'에 샘플 데이터를 추가합니다. 이 예시에서는 하나의 레코드만 추가했습니다.
  4. 데이터베이스 연결을 종료합니다.

위의 예시 코드를 실행하면 'mydb.db' 파일이 생성되고, 'mytable' 테이블이 생성되며, 샘플 데이터가 추가됩니다. 이제 앞서 작성한 Node.js와 EJS 코드에서 해당 데이터베이스를 사용하여 제목, 설명, 제작자 정보를 가져와서 렌더링할 수 있을 것입니다.

위의 코드를 실행하고 http://localhost:3000/cms에 접속하면 데이터 입력 페이지인 cms.html이 나타납니다. 사용자가 데이터를 입력하고 저장하면 SQLite 데이터베이스에 저장되고, 메인 페이지(http://localhost:3000/)에서 변경된 데이터를 확인할 수 있습니다.
